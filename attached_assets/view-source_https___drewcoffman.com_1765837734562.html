// Theme toggle
function toggleTheme() {
const wrapper = document.getElementById('page-wrapper');
const isDark = wrapper.classList.contains('dark-mode');
if (isDark) {
wrapper.classList.remove('dark-mode');
localStorage.setItem('theme', 'light');
updateThemeIcon('light');
} else {
wrapper.classList.add('dark-mode');
localStorage.setItem('theme', 'dark');
updateThemeIcon('dark');
}
}

function updateThemeIcon(theme) {
const icon = document.querySelector('.theme-icon');
icon.textContent = theme === 'dark' ? '☽' : '☀';
}

function loadTheme() {
const savedTheme = localStorage.getItem('theme') || 'light';
if (savedTheme === 'dark') {
document.getElementById('page-wrapper').classList.add('dark-mode');
}
updateThemeIcon(savedTheme);
}

div.appendChild(badge);

const img = document.createElement('img');

img.alt = book.title;
img.loading = 'lazy';
img.onload = function() {

const grid = document.getElementById(`movies-${year}`);
if (grid) {
movies[year].forEach(movie => {
grid.appendChild(createMovieElement(movie));
});
}
});
}

// Essays
let currentEssaySort = 'all';

function sortEssays(type) {
currentEssaySort = type;
document.querySelectorAll('#tab-essays .sort-btn').forEach(btn => {
btn.classList.remove('active');
});
event.target.classList.add('active');
populateEssays();
updateEssayCount();
}

function updateEssayCount() {
const list = document.getElementById('essays-list');
const count = list ? list.children.length : 0;
document.getElementById('essay-count').textContent = `${count} essays`;
}

function createEssayElement(essay) {
const li = document.createElement('li');
li.className = 'essay-item';
const titleDiv = document.createElement('div');
titleDiv.className = 'essay-title';
const link = document.createElement('a');
link.href = essay.url;
link.target = '_blank';
link.textContent = essay.title;
titleDiv.appendChild(link);
if (essay.badge) {
const badge = document.createElement('span');
badge.className = 'essay-badge';
if (essay.badge === '❤') badge.classList.add('heart');
if (essay.badge === '★') badge.classList.add('star');
badge.textContent = essay.badge;
titleDiv.appendChild(badge);
}
li.appendChild(titleDiv);
const meta = document.createElement('div');
meta.className = 'essay-meta';
meta.textContent = `${essay.author} · ${essay.source}`;
li.appendChild(meta);
if (essay.review) {
const review = document.createElement('div');
review.className = 'essay-review';
review.textContent = essay.review;
li.appendChild(review);
}
return li;
}

function populateEssays() {
const list = document.getElementById('essays-list');
if (list) {
list.innerHTML = '';
let filteredEssays = essays;
if (currentEssaySort === 'top') {
filteredEssays = essays.filter(e => e.badge === '★');
} else if (currentEssaySort === 'favorites') {
filteredEssays = essays.filter(e => e.badge === '❤');
}
filteredEssays.forEach(essay => {
list.appendChild(createEssayElement(essay));
});
}
}
